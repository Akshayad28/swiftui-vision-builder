SELECT
    a.landid,
    a.business_date,
    a.job_id,
    b.producttype,
    c.uklongclassification,
    b.holding_classification,
    b.lei,

    CASE
        WHEN EXISTS (
            SELECT 1
            FROM rpt_cat r
            WHERE r.holding_classification = b.holding_classification
        )
        THEN b.holding_classification
        ELSE 'non exempt'
    END AS final_holding_classification

FROM eligibility a

JOIN stage b
  ON a.landid = b.landid
 AND a.business_date = b.business_date

LEFT JOIN (
    SELECT DISTINCT
        producttype,
        uklongclassification
    FROM product
) c
  ON b.producttype = c.producttype

WHERE a.business_date = '2026-04-20'
  AND a.job_id IN ('JOB1', 'JOB2', 'JOB3')
  AND b.lei = 'YOUR_LEI';
